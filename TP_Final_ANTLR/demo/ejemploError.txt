// Ejemplo para probar todos los warnings

// Declaración de variables globales
int global = 10;
double unused_var = 3.14;  // Warning: variable declarada pero nunca usada

// Función principal
int main() {
    // Warning: variable declarada pero nunca usada
    char letter = 'A';
    
    // Warning: variable usada sin inicializar
    int x;
    int result = x + 5;  // x no fue inicializada
    
    // Warning: conversión implícita de tipos
    double pi = 3.14;
    int truncated = pi;  // double -> int (pérdida de precisión)
    
    // Warning: división por cero
    int division = 10 / 0;
    
    // Warning: asignación en lugar de comparación
    int y = 5;
    if (y = 10) {  // Debería ser y == 10
        y = 20;
    }
    
    // Código válido que se ejecuta
    int z = 15;
    z = z + 1;
    
    // Warning: código inalcanzable
    return 0;
    
    // Todo lo siguiente es código inalcanzable
    int dead_code = 999;  // Warning: código después de return
    z = z + 100;          // Warning: código después de return
    
    if (z > 10) {         // Warning: código después de return
        z = 0;
    }
}
